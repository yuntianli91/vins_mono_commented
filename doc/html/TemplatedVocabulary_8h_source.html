638 template<class TDescriptor, class F>
639 void TemplatedVocabulary<TDescriptor,F>::HKmeansStep(NodeId parent_id,
640  const std::vector<pDescriptor> &descriptors, int current_level)
641 {
642  if(descriptors.empty()) return;
643 
644  // features associated to each cluster
645  std::vector<TDescriptor> clusters;
646  std::vector<std::vector<unsigned int> > groups; // groups[i] = [j1, j2, ...]
647  // j1, j2, ... indices of descriptors associated to cluster i
648 
649  clusters.reserve(m_k);
650  groups.reserve(m_k);
651 
652  //const int msizes[] = { m_k, descriptors.size() };
653  //cv::SparseMat assoc(2, msizes, CV_8U);
654  //cv::SparseMat last_assoc(2, msizes, CV_8U);
656 
657  if((int)descriptors.size() <= m_k)
658  {
659  // trivial case: one cluster per feature
660  groups.resize(descriptors.size());
661 
662  for(unsigned int i = 0; i < descriptors.size(); i++)
663  {
664  groups[i].push_back(i);
665  clusters.push_back(*descriptors[i]);
666  }
667  }
668  else
669  {
670  // select clusters and groups with kmeans
671 
672  bool first_time = true;
673  bool goon = true;
674 
675  // to check if clusters move after iterations
676  std::vector<int> last_association, current_association;
677 
678  while(goon)
679  {
680  // 1. Calculate clusters
681 
682  if(first_time)
683  {
684  // random sample
685  initiateClusters(descriptors, clusters);
686  }
687  else
688  {
689  // calculate cluster centres
690 
691  for(unsigned int c = 0; c < clusters.size(); ++c)
692  {
693  std::vector<pDescriptor> cluster_descriptors;
694  cluster_descriptors.reserve(groups[c].size());
695 
696  /*
697  for(unsigned int d = 0; d < descriptors.size(); ++d)
698  {
699  if( assoc.find<unsigned char>(c, d) )
700  {
701  cluster_descriptors.push_back(descriptors[d]);
702  }
703  }
704  */
705 
706  std::vector<unsigned int>::const_iterator vit;
707  for(vit = groups[c].begin(); vit != groups[c].end(); ++vit)
708  {
709  cluster_descriptors.push_back(descriptors[*vit]);
710  }
711 
712 
713  F::meanValue(cluster_descriptors, clusters[c]);
714  }
715 
716  } // if(!first_time)
717 
718  // 2. Associate features with clusters
719 
720  // calculate distances to cluster centers
721  groups.clear();
722  groups.resize(clusters.size(), std::vector<unsigned int>());
723  current_association.resize(descriptors.size());
724 
725  //assoc.clear();
726 
727  typename std::vector<pDescriptor>::const_iterator fit;
728  //unsigned int d = 0;
729  for(fit = descriptors.begin(); fit != descriptors.end(); ++fit)//, ++d)
730  {
731  double best_dist = F::distance(*(*fit), clusters[0]);
732  unsigned int icluster = 0;
733 
734  for(unsigned int c = 1; c < clusters.size(); ++c)
735  {
736  double dist = F::distance(*(*fit), clusters[c]);
737  if(dist < best_dist)
738  {
739  best_dist = dist;
740  icluster = c;
741  }
742  }
743 
744  //assoc.ref<unsigned char>(icluster, d) = 1;
745 
746  groups[icluster].push_back(fit - descriptors.begin());
747  current_association[ fit - descriptors.begin() ] = icluster;
748  }
749 
750  // kmeans++ ensures all the clusters has any feature associated with them
751 
752  // 3. check convergence
753  if(first_time)
754  {
755  first_time = false;
756  }
757  else
758  {
759  //goon = !eqUChar(last_assoc, assoc);
760 
761  goon = false;
762  for(unsigned int i = 0; i < current_association.size(); i++)
763  {
764  if(current_association[i] != last_association[i]){
765  goon = true;
766  break;
767  }
768  }
769  }
770 
771  if(goon)
772  {
773  // copy last feature-cluster association
774  last_association = current_association;
775  //last_assoc = assoc.clone();
776  }
777 
778  } // while(goon)
779 
780  } // if must run kmeans
781 
782  // create nodes
783  for(unsigned int i = 0; i < clusters.size(); ++i)
784  {
785  NodeId id = m_nodes.size();
786  m_nodes.push_back(Node(id));
787  m_nodes.back().descriptor = clusters[i];
788  m_nodes.back().parent = parent_id;
789  m_nodes[parent_id].children.push_back(id);
790  }
791 
792  // go on with the next level
793  if(current_level < m_L)
794  {
795  // iterate again with the resulting clusters
796  const std::vector<NodeId> &children_ids = m_nodes[parent_id].children;
797  for(unsigned int i = 0; i < clusters.size(); ++i)
798  {
799  NodeId id = children_ids[i];
800 
801  std::vector<pDescriptor> child_features;
802  child_features.reserve(groups[i].size());
803 
804  std::vector<unsigned int>::const_iterator vit;
805  for(vit = groups[i].begin(); vit != groups[i].end(); ++vit)
806  {
807  child_features.push_back(descriptors[*vit]);
808  }
809 
810  if(child_features.size() > 1)
811  {
812  HKmeansStep(id, child_features, current_level + 1);
813  }
814  }
815  }
816 }

830 template<class TDescriptor, class F>
831 void TemplatedVocabulary<TDescriptor,F>::initiateClustersKMpp(
832  const std::vector<pDescriptor> &pfeatures,
833  std::vector<TDescriptor> &clusters) const
834 {
835  // Implements kmeans++ seeding algorithm
836  // Algorithm:
837  // 1. Choose one center uniformly at random from among the data points.
838  // 2. For each data point x, compute D(x), the distance between x and the nearest
839  // center that has already been chosen.
840  // 3. Add one new data point as a center. Each point x is chosen with probability
841  // proportional to D(x)^2.
842  // 4. Repeat Steps 2 and 3 until k centers have been chosen.
843  // 5. Now that the initial centers have been chosen, proceed using standard k-means
844  // clustering.
845 
846  DUtils::Random::SeedRandOnce();
847 
848  clusters.resize(0);
849  clusters.reserve(m_k);
850  std::vector<double> min_dists(pfeatures.size(), std::numeric_limits<double>::max());
851 
852  // 1.
853 
854  int ifeature = DUtils::Random::RandomInt(0, pfeatures.size()-1);
855 
856  // create first cluster
857  clusters.push_back(*pfeatures[ifeature]);
858 
859  // compute the initial distances
860  typename std::vector<pDescriptor>::const_iterator fit;
861  std::vector<double>::iterator dit;
862  dit = min_dists.begin();
863  for(fit = pfeatures.begin(); fit != pfeatures.end(); ++fit, ++dit)
864  {
865  *dit = F::distance(*(*fit), clusters.back());
866  }
867 
868  while((int)clusters.size() < m_k)
869  {
870  // 2.
871  dit = min_dists.begin();
872  for(fit = pfeatures.begin(); fit != pfeatures.end(); ++fit, ++dit)
873  {
874  if(*dit > 0)
875  {
876  double dist = F::distance(*(*fit), clusters.back());
877  if(dist < *dit) *dit = dist;
878  }
879  }
880 
881  // 3.
882  double dist_sum = std::accumulate(min_dists.begin(), min_dists.end(), 0.0);
883 
884  if(dist_sum > 0)
885  {
886  double cut_d;
887  do
888  {
889  cut_d = DUtils::Random::RandomValue<double>(0, dist_sum);
890  } while(cut_d == 0.0);
891 
892  double d_up_now = 0;
893  for(dit = min_dists.begin(); dit != min_dists.end(); ++dit)
894  {
895  d_up_now += *dit;
896  if(d_up_now >= cut_d) break;
897  }
898 
899  if(dit == min_dists.end())
900  ifeature = pfeatures.size()-1;
901  else
902  ifeature = dit - min_dists.begin();
903 
904  clusters.push_back(*pfeatures[ifeature]);
905 
906  } // if dist_sum > 0
907  else
908  break;
909 
910  } // while(used_clusters < m_k)
911 
912 }

941 template<class TDescriptor, class F>
942 void TemplatedVocabulary<TDescriptor,F>::setNodeWeights
943  (const std::vector<std::vector<TDescriptor> > &training_features)
944 {
945  const unsigned int NWords = m_words.size();
946  const unsigned int NDocs = training_features.size();
947 
948  if(m_weighting == TF || m_weighting == BINARY)
949  {
950  // idf part must be 1 always
951  for(unsigned int i = 0; i < NWords; i++)
952  m_words[i]->weight = 1;
953  }
954  else if(m_weighting == IDF || m_weighting == TF_IDF)
955  {
956  // IDF and TF-IDF: we calculte the idf path now
957 
958  // Note: this actually calculates the idf part of the tf-idf score.
959  // The complete tf-idf score is calculated in ::transform
960 
961  std::vector<unsigned int> Ni(NWords, 0);
962  std::vector<bool> counted(NWords, false);
963 
964  typename std::vector<std::vector<TDescriptor> >::const_iterator mit;
965  typename std::vector<TDescriptor>::const_iterator fit;
966 
967  for(mit = training_features.begin(); mit != training_features.end(); ++mit)
968  {
969  fill(counted.begin(), counted.end(), false);
970 
971  for(fit = mit->begin(); fit < mit->end(); ++fit)
972  {
973  WordId word_id;
974  transform(*fit, word_id);
975 
976  if(!counted[word_id])
977  {
978  Ni[word_id]++;
979  counted[word_id] = true;
980  }
981  }
982  }
983 
984  // set ln(N/Ni)
985  for(unsigned int i = 0; i < NWords; i++)
986  {
987  if(Ni[i] > 0)
988  {
989  m_words[i]->weight = log((double)NDocs / (double)Ni[i]);
990  }// else // This cannot occur if using kmeans++
991  }
992 
993  }
994

1216 template<class TDescriptor, class F>
1217 void TemplatedVocabulary<TDescriptor,F>::transform(const TDescriptor &feature,
1218  WordId &word_id, WordValue &weight, NodeId *nid, int levelsup) const
1219 {
1220  // propagate the feature down the tree
1221  std::vector<NodeId> nodes;
1222  typename std::vector<NodeId>::const_iterator nit;
1223 
1224  // level at which the node must be stored in nid, if given
1225  const int nid_level = m_L - levelsup;
1226  if(nid_level <= 0 && nid != NULL) *nid = 0; // root
1227 
1228  NodeId final_id = 0; // root
1229  int current_level = 0;
1230 
1231  do
1232  {
1233  ++current_level;
1234  nodes = m_nodes[final_id].children;
1235  final_id = nodes[0];
1236 
1237  double best_d = F::distance(feature, m_nodes[final_id].descriptor);
1238 
1239  for(nit = nodes.begin() + 1; nit != nodes.end(); ++nit)
1240  {
1241  NodeId id = *nit;
1242  double d = F::distance(feature, m_nodes[id].descriptor);
1243  if(d < best_d)
1244  {
1245  best_d = d;
1246  final_id = id;
1247  }
1248  }
1249 
1250  if(nid != NULL && current_level == nid_level)
1251  *nid = final_id;
1252 
1253  } while( !m_nodes[final_id].isLeaf() );
1254 
1255  // turn node id into word id
1256  word_id = m_nodes[final_id].word_id;
1257  weight = m_nodes[final_id].weight;
